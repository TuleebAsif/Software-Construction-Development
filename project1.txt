      <script>

          function addNumbers() {
              let num1 = parseInt(document.getElementById("num1").value);
              let num2 = parseInt(document.getElementById("num2").value);
              let result = document.getElementById("sum");
              result.innerHTML = num1 + num2;
          }




          function greet() {
              let name = document.getElementById("name").value;
              let greeting = document.getElementById("greeting");
              greeting.innerHTML = "Hello, " + name + "!";
          }


          function findAverage() {
              let numbers = document.getElementById("numbers").value;
              let numberArray = numbers.split(",").map(Number);
              let sum = numberArray.reduce((acc, curr) => acc + curr);
              let average = sum / numberArray.length;
              let result = document.getElementById("average");
              result.innerHTML = "The average is " + average.toFixed(2);
          }


          function checkEvenOdd() {
              let number = parseInt(document.getElementById("number").value);
              let result = document.getElementById("even-odd");
              if (number % 2 == 0) {
                  result.innerHTML = number + " is even.";
              } else {
                  result.innerHTML = number + " is odd.";
              }
          }


          function reverseString() {
              let string = document.getElementById("string").value;
              let reversed = string.split("").reverse().join("");
              let result = document.getElementById("reversed");
              result.innerHTML = "The reversed string is " + reversed;
          }


          function guessNumber() {
              let randomNumber = Math.floor(Math.random() * 10) + 1;
              let guess = parseInt(document.getElementById("guess").value);
              let result = document.getElementById("guess-result");
              if (guess == randomNumber) {
                  result.innerHTML = "Congratulations! You guessed the number.";
              } else if (guess < randomNumber) {
                  result.innerHTML = "Your guess is too low. Try again.";
              } else {
                  result.innerHTML = "Your guess is too high. Try again.";
              }
          }


          function checkPrime() {
              let number = parseInt(document.getElementById("prime").value);
              let result = document.getElementById("prime-result");
              if (number <= 1) {
                  result.innerHTML = number + " is not prime.";
                  return;
              }
              for (let i = 2; i <= Math.sqrt(number); i++) {
                  if (number % i == 0) {
                      result.innerHTML = number + " is not prime.";
                      return;
                  }
              }
              result.innerHTML = number + " is prime.";
          }


          function countWords() {
              let sentence = document.getElementById("sentence").value;
              let words = sentence.split(" ").filter((word) => word != "").length;
              let result = document.getElementById("word-count");
              result.innerHTML = "The sentence has " + words + " words.";
          }


          function findLongest() {
              let strings = document.getElementById("strings").value;
              let stringArray = strings.split(",");
              let longest = stringArray.reduce(function (a, b) {
                  return a.length > b.length ? a : b;
              });
              let result = document.getElementById("longest");
              result.innerHTML = "The longest string is " + longest;
          }


          function checkVoting() {
              let age = parseInt(document.getElementById("age").value);
              let result = document.getElementById("voting-result");
              if (age >= 18) {
                  result.innerHTML = "You are eligible to vote!";
              } else {
                  result.innerHTML = "You are not eligible to vote!";
              }

          }
